```bash
export AWS_REGION=ap-northeast-2

export ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
export AZ_A=$(aws ec2 describe-availability-zones --query 'AvailabilityZones[0].ZoneName' --output text)

echo "Account=$ACCOUNT_ID  AZ_A=$AZ_A  AZ_B=$AZ_B"
```


### AWS CLI — EKS Cluster Role

**Bash / Git Bash / WSL** ( `\` = 줄 이어쓰기):

```bash
aws iam create-role \
  --role-name forenshield-eks-cluster-role \
  --assume-role-policy-document '{
    "Version": "2012-10-17",
    "Statement": [{
      "Effect": "Allow",
      "Principal": { "Service": "eks.amazonaws.com" },
      "Action": "sts:AssumeRole"
    }]
  }'

aws iam attach-role-policy \
  --role-name forenshield-eks-cluster-role \
  --policy-arn arn:aws:iam::aws:policy/AmazonEKSClusterPolicy
```

**Windows PowerShell** (JSON은 **파일로 저장** — 인라인 JSON은 `MalformedPolicyDocument` 자주 발생):

```powershell
# 1) Trust policy JSON 파일 생성 (BOM 없이)
$trust = @'
{
  "Version": "2012-10-17",
  "Statement": [{
    "Effect": "Allow",
    "Principal": { "Service": "eks.amazonaws.com" },
    "Action": "sts:AssumeRole"
  }]
}
'@
$policyPath = "$env:TEMP\trust-eks-cluster.json"
[System.IO.File]::WriteAllText($policyPath, $trust)

# 2) Role 생성 (Windows는 file:// + 절대경로)
$policyUri = "file://" + ($policyPath -replace '\\', '/')
aws iam create-role `
  --role-name forenshield-eks-cluster-role `
  --assume-role-policy-document $policyUri

aws iam attach-role-policy `
  --role-name forenshield-eks-cluster-role `
  --policy-arn arn:aws:iam::aws:policy/AmazonEKSClusterPolicy
```

> PowerShell에서 `'{"Version":...}'` **한 줄 JSON**은 따옴표·인코딩 때문에 AWS가 `invalid Json` 으로 거절하는 경우가 많습니다. **반드시 파일 방식**을 쓰세요.

## Step 2 — VPC · Subnet · Internet Gateway

| 리소스 | 설정 |
|--------|------|
| VPC | `10.0.0.0/16` |
| Public Subnet × 2 | AZ-a, AZ-b — ALB, NAT 전용 |
| Private Subnet × 2 | AZ-a, AZ-b — EKS Worker (Frontend/Backend/AI) |
| Data Subnet × 2 | AZ-a, AZ-b — RDS, ElastiCache |
| Internet Gateway | Public Subnet 연결 |

### AWS CLI — VPC · Subnet · IGW

> **Subnet 만들기 전 필수:** `$AZ_A`, `$AZ_B` 가 비어 있으면  
> `argument --availability-zone: expected one argument` 오류가 납니다.

```bash
# ★ Subnet 전에 반드시 실행 (공통 설정을 건너뛴 경우)
export ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
export AZ_A=$(aws ec2 describe-availability-zones --region ap-northeast-2 \
  --query 'AvailabilityZones[0].ZoneName' --output text)
export AZ_B=$(aws ec2 describe-availability-zones --region ap-northeast-2 \
  --query 'AvailabilityZones[1].ZoneName' --output text)
echo "AZ_A=$AZ_A  AZ_B=$AZ_B"   # 예: ap-northeast-2a  ap-northeast-2b

# VPC
export VPC_ID=$(aws ec2 create-vpc \
  --cidr-block 10.0.0.0/16 \
  --tag-specifications 'ResourceType=vpc,Tags=[{Key=Name,Value=forenshield-vpc},{Key=Project,Value=forenshield}]' \
  --query 'Vpc.VpcId' --output text)

aws ec2 modify-vpc-attribute --vpc-id $VPC_ID --enable-dns-hostnames
aws ec2 modify-vpc-attribute --vpc-id $VPC_ID --enable-dns-support

# Internet Gateway
export IGW_ID=$(aws ec2 create-internet-gateway \
  --tag-specifications 'ResourceType=internet-gateway,Tags=[{Key=Name,Value=forenshield-igw}]' \
  --query 'InternetGateway.InternetGatewayId' --output text)
aws ec2 attach-internet-gateway --internet-gateway-id $IGW_ID --vpc-id $VPC_ID

# Public Subnets
export PUB_SUBNET_A=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.1.0/24 \
  --availability-zone $AZ_A \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-public-a}]' \
  --query 'Subnet.SubnetId' --output text)
export PUB_SUBNET_B=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.2.0/24 \
  --availability-zone $AZ_B \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-public-b}]' \
  --query 'Subnet.SubnetId' --output text)
aws ec2 modify-subnet-attribute --subnet-id $PUB_SUBNET_A --map-public-ip-on-launch
aws ec2 modify-subnet-attribute --subnet-id $PUB_SUBNET_B --map-public-ip-on-launch

# Private Subnets (EKS)
export PRIV_SUBNET_A=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.10.0/24 \
  --availability-zone $AZ_A \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-private-a}]' \
  --query 'Subnet.SubnetId' --output text)
export PRIV_SUBNET_B=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.11.0/24 \
  --availability-zone $AZ_B \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-private-b}]' \
  --query 'Subnet.SubnetId' --output text)

# Data Subnets (RDS / Redis)
export DATA_SUBNET_A=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.20.0/24 \
  --availability-zone $AZ_A \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-data-a}]' \
  --query 'Subnet.SubnetId' --output text)
export DATA_SUBNET_B=$(aws ec2 create-subnet --vpc-id $VPC_ID --cidr-block 10.0.21.0/24 \
  --availability-zone $AZ_B \
  --tag-specifications 'ResourceType=subnet,Tags=[{Key=Name,Value=forenshield-data-b}]' \
  --query 'Subnet.SubnetId' --output text)

# Public Route Table
export PUB_RT=$(aws ec2 create-route-table --vpc-id $VPC_ID \
  --tag-specifications 'ResourceType=route-table,Tags=[{Key=Name,Value=forenshield-public-rt}]' \
  --query 'RouteTable.RouteTableId' --output text)
aws ec2 create-route --route-table-id $PUB_RT --destination-cidr-block 0.0.0.0/0 --gateway-id $IGW_ID
aws ec2 associate-route-table --route-table-id $PUB_RT --subnet-id $PUB_SUBNET_A
aws ec2 associate-route-table --route-table-id $PUB_RT --subnet-id $PUB_SUBNET_B

echo "VPC_ID=$VPC_ID"
```

### 검증

```bash
aws ec2 describe-vpcs --filters "Name=tag:Project,Values=forenshield" \
  --query 'Vpcs[*].[VpcId,CidrBlock,State]' --output table

aws ec2 describe-subnets --filters "Name=vpc-id,Values=$VPC_ID" \
  --query 'Subnets[*].[SubnetId,CidrBlock,AvailabilityZone,Tags[?Key==`Name`].Value|[0]]' --output table
```

## Step 3 — NAT Gateway

| 리소스 | 설정 |
|--------|------|
| NAT Gateway | Public Subnet 1개 (비용 절감) |
| Private Route Table | `0.0.0.0/0` → NAT Gateway |

### AWS CLI

```bash
# Elastic IP
export NAT_EIP=$(aws ec2 allocate-address --domain vpc \
  --tag-specifications 'ResourceType=elastic-ip,Tags=[{Key=Name,Value=forenshield-nat-eip}]' \
  --query 'AllocationId' --output text)

# NAT Gateway (Public Subnet A 1개 — 비용 절감)
export NAT_GW=$(aws ec2 create-nat-gateway --subnet-id $PUB_SUBNET_A --allocation-id $NAT_EIP \
  --tag-specifications 'ResourceType=natgateway,Tags=[{Key=Name,Value=forenshield-nat}]' \
  --query 'NatGateway.NatGatewayId' --output text)

# NAT active 대기 (1~2분)
aws ec2 wait nat-gateway-available --nat-gateway-ids $NAT_GW

# Private Route Table
export PRIV_RT=$(aws ec2 create-route-table --vpc-id $VPC_ID \
  --tag-specifications 'ResourceType=route-table,Tags=[{Key=Name,Value=forenshield-private-rt}]' \
  --query 'RouteTable.RouteTableId' --output text)
aws ec2 create-route --route-table-id $PRIV_RT --destination-cidr-block 0.0.0.0/0 --nat-gateway-id $NAT_GW
aws ec2 associate-route-table --route-table-id $PRIV_RT --subnet-id $PRIV_SUBNET_A
aws ec2 associate-route-table --route-table-id $PRIV_RT --subnet-id $PRIV_SUBNET_B

# Data Route Table (인터넷 불필요 — S3 VPC Endpoint 사용)
export DATA_RT=$(aws ec2 create-route-table --vpc-id $VPC_ID \
  --tag-specifications 'ResourceType=route-table,Tags=[{Key=Name,Value=forenshield-data-rt}]' \
  --query 'RouteTable.RouteTableId' --output text)
aws ec2 associate-route-table --route-table-id $DATA_RT --subnet-id $DATA_SUBNET_A
aws ec2 associate-route-table --route-table-id $DATA_RT --subnet-id $DATA_SUBNET_B
```

### 검증

```bash
aws ec2 describe-nat-gateways --nat-gateway-ids $NAT_GW \
  --query 'NatGateways[*].[NatGatewayId,State,SubnetId]' --output table

aws ec2 describe-route-tables --route-table-ids $PRIV_RT \
  --query 'RouteTables[0].Routes' --output table
```

## Step 4 — Security Group

| SG 이름 | 인바운드 | 아웃바운드 |
|---------|----------|------------|
| `sg-alb` | 80, 443 from `0.0.0.0/0` | ALL |
| `sg-eks-node` | ALB SG → 80, 8080 | ALL |
| `sg-rds` | EKS Node SG → 5432 | — |
| `sg-redis` | EKS Node SG → 6379 | — |
| `sg-vpn` | On-Prem CIDR | ALL |

### AWS CLI

```bash
# sg-alb
export SG_ALB=$(aws ec2 create-security-group --group-name forenshield-sg-alb \
  --description "ALB HTTP/HTTPS" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_ALB --protocol tcp --port 80 --cidr 0.0.0.0/0
aws ec2 authorize-security-group-ingress --group-id $SG_ALB --protocol tcp --port 443 --cidr 0.0.0.0/0

# sg-eks-node
export SG_EKS=$(aws ec2 create-security-group --group-name forenshield-sg-eks-node \
  --description "EKS worker nodes" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_EKS --protocol tcp --port 80 \
  --source-group $SG_ALB
aws ec2 authorize-security-group-ingress --group-id $SG_EKS --protocol tcp --port 8080 \
  --source-group $SG_ALB
aws ec2 authorize-security-group-ingress --group-id $SG_EKS --protocol -1 \
  --source-group $SG_EKS

# sg-rds
export SG_RDS=$(aws ec2 create-security-group --group-name forenshield-sg-rds \
  --description "RDS PostgreSQL" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_RDS --protocol tcp --port 5432 \
  --source-group $SG_EKS

# sg-redis
export SG_REDIS=$(aws ec2 create-security-group --group-name forenshield-sg-redis \
  --description "ElastiCache Redis" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_REDIS --protocol tcp --port 6379 \
  --source-group $SG_EKS

# sg-vpn (On-Prem CIDR — 예: 192.168.0.0/24)
export ONPREM_CIDR=192.168.0.0/24
export SG_VPN=$(aws ec2 create-security-group --group-name forenshield-sg-vpn \
  --description "Site-to-Site VPN" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_VPN --protocol -1 --cidr $ONPREM_CIDR
```

### 검증

```bash
aws ec2 describe-security-groups --filters "Name=vpc-id,Values=$VPC_ID" \
  --query 'SecurityGroups[?contains(GroupName,`forenshield`)].[GroupName,GroupId]' --output table
```

## Step 5 — VPC Endpoint

| Endpoint | 용도 |
|----------|------|
| S3 Gateway Endpoint | Private Subnet → S3 (증거·모델 버킷) |
| ECR API / DKR Interface Endpoint | NAT 없이 이미지 pull (선택, 비용 고려) |

### AWS CLI — S3 Gateway Endpoint

```bash
export VPCE_S3=$(aws ec2 create-vpc-endpoint \
  --vpc-id $VPC_ID \
  --service-name com.amazonaws.ap-northeast-2.s3 \
  --route-table-ids $PRIV_RT $DATA_RT \
  --query 'VpcEndpoint.VpcEndpointId' --output text)

aws ec2 describe-vpc-endpoints --vpc-endpoint-ids $VPCE_S3 \
  --query 'VpcEndpoints[0].[VpcEndpointId,State,ServiceName]' --output table
```

### AWS CLI — ECR Interface Endpoint (선택, 비용 발생)

```bash
# Interface Endpoint용 SG
export SG_VPCE=$(aws ec2 create-security-group --group-name forenshield-sg-vpce \
  --description "VPC Interface Endpoints" --vpc-id $VPC_ID --query 'GroupId' --output text)
aws ec2 authorize-security-group-ingress --group-id $SG_VPCE --protocol tcp --port 443 \
  --source-group $SG_EKS

aws ec2 create-vpc-endpoint --vpc-id $VPC_ID \
  --vpc-endpoint-type Interface \
  --service-name com.amazonaws.ap-northeast-2.ecr.api \
  --subnet-ids $PRIV_SUBNET_A $PRIV_SUBNET_B \
  --security-group-ids $SG_VPCE

aws ec2 create-vpc-endpoint --vpc-id $VPC_ID \
  --vpc-endpoint-type Interface \
  --service-name com.amazonaws.ap-northeast-2.ecr.dkr \
  --subnet-ids $PRIV_SUBNET_A $PRIV_SUBNET_B \
  --security-group-ids $SG_VPCE
```

## Step 6 — S3 버킷

| 버킷 | 설정 |

### AWS CLI

> S3 버킷명은 **전 세계 유일**해야 합니다. 아래는 `계정ID` 접미사 예시입니다.


## Step 9 — Site-to-Site VPN (On-Prem GPU 연동)

Phase 1에서 검증한 On-Prem IP·CIDR을 사용합니다.

| 리소스 | 설정 |
|--------|------|
| Customer Gateway | On-Prem 공인 IP |
| Virtual Private Gateway | VPC 연결 |
| VPN Connection | IPSec 터널 × 2 |
| Route Table | On-Prem CIDR → VGW |

### AWS CLI

```bash
export ONPREM_PUBLIC_IP=203.0.113.10   # Phase 1 GPU 서버 공인 IP
export ONPREM_CIDR=192.168.0.0/24

# Customer Gateway
export CGW_ID=$(aws ec2 create-customer-gateway \
  --type ipsec.1 --bgp-asn 65000 --public-ip $ONPREM_PUBLIC_IP \
  --tag-specifications 'ResourceType=customer-gateway,Tags=[{Key=Name,Value=forenshield-cgw}]' \
  --query 'CustomerGateway.CustomerGatewayId' --output text)

# Virtual Private Gateway
export VGW_ID=$(aws ec2 create-vpn-gateway --type ipsec.1 \
  --tag-specifications 'ResourceType=vpn-gateway,Tags=[{Key=Name,Value=forenshield-vgw}]' \
  --query 'VpnGateway.VpnGatewayId' --output text)
aws ec2 attach-vpn-gateway --vpn-gateway-id $VGW_ID --vpc-id $VPC_ID

# VPN Connection
export VPN_ID=$(aws ec2 create-vpn-connection \
  --type ipsec.1 --customer-gateway-id $CGW_ID --vpn-gateway-id $VGW_ID \
  --options StaticRoutesOnly=true \
  --tag-specifications 'ResourceType=vpn-connection,Tags=[{Key=Name,Value=forenshield-vpn}]' \
  --query 'VpnConnection.VpnConnectionId' --output text)

aws ec2 create-vpn-connection-route \
  --vpn-connection-id $VPN_ID --destination-cidr-block $ONPREM_CIDR

# Private · Data Route Table에 VGW 전파
aws ec2 enable-vgw-route-propagation --route-table-id $PRIV_RT --gateway-id $VGW_ID
aws ec2 enable-vgw-route-propagation --route-table-id $DATA_RT --gateway-id $VGW_ID

# On-Prem 방화벽 설정용 터널 정보 출력
aws ec2 describe-vpn-connections --vpn-connection-ids $VPN_ID \
  --query 'VpnConnections[0].VgwTelemetry' --output table
```

### 개요

| 항목 | 값 |
|------|------|
| IaC | Terraform `helm_release` (Bitnami Chart) |
| Namespace | `forenshield` |
| Node Group | `backend-ng` (`nodeSelector`) |
| 서비스 DNS | `rabbitmq.forenshield.svc.cluster.local:5672` |
| 스토리지 | EBS PVC 10Gi (`StorageClass: gp3`) |
| HA | Phase 1 — replica 1 |

### Terraform 파일 구조

```
terraform/
├── providers.tf          # kubernetes, helm provider (EKS kubeconfig 연동)
├── eks.tf                # Step 10 — cluster, namespace
├── eks-nodegroups.tf     # Step 11 — backend-ng
├── ebs-storage.tf        # gp3 StorageClass (EBS CSI Add-on 필요)
└── rabbitmq.tf           # Step 11-1 — RabbitMQ Helm 배포
```

### Provider 설정

EKS 클러스터 생성 후 `kubernetes` / `helm` provider가 클러스터 API를 바라보도록 설정한다.

```hcl
# providers.tf
data "aws_eks_cluster" "this" {
  name = var.cluster_name
}

data "aws_eks_cluster_auth" "this" {
  name = var.cluster_name
}

provider "kubernetes" {
  host                   = data.aws_eks_cluster.this.endpoint
  cluster_ca_certificate = base64decode(data.aws_eks_cluster.this.certificate_authority[0].data)
  token                  = data.aws_eks_cluster_auth.this.token
}

provider "helm" {
  kubernetes {
    host                   = data.aws_eks_cluster.this.endpoint
    cluster_ca_certificate = base64decode(data.aws_eks_cluster.this.certificate_authority[0].data)
    token                  = data.aws_eks_cluster_auth.this.token
  }
}
```

### gp3 StorageClass (PVC 선행)

RabbitMQ Helm Chart는 PVC가 필요하므로, EBS CSI Add-on 설치 후 StorageClass를 Terraform으로 만든다.

```hcl
# ebs-storage.tf
resource "kubernetes_storage_class_v1" "gp3" {
  metadata {
    name = "gp3"
  }
  storage_provisioner    = "ebs.csi.aws.com"
  reclaim_policy         = "Retain"
  volume_binding_mode    = "WaitForFirstConsumer"
  allow_volume_expansion = true

  parameters = {
    type = "gp3"
  }
}
```

> EKS Add-on: `aws-ebs-csi-driver` 가 Step 10에서 함께 설치되어 있어야 한다.

### RabbitMQ Terraform

```hcl
# rabbitmq.tf
resource "random_password" "rabbitmq" {
  length  = 24
  special = false
}

resource "helm_release" "rabbitmq" {
  name       = "rabbitmq"
  repository = "https://charts.bitnami.com/bitnami"
  chart      = "rabbitmq"
  namespace  = kubernetes_namespace.forenshield.metadata[0].name
  version    = "14.6.9" # chart 버전 고정

  values = [
    yamlencode({
      auth = {
        username = var.rabbitmq_username
        password = random_password.rabbitmq.result
      }
      persistence = {
        enabled      = true
        size         = "10Gi"
        storageClass = kubernetes_storage_class_v1.gp3.metadata[0].name
      }
      resources = {
        requests = { memory = "512Mi", cpu = "250m" }
        limits   = { memory = "1Gi", cpu = "500m" }
      }
      nodeSelector = {
        "eks.amazonaws.com/nodegroup" = "backend-ng"
      }
      service = {
        type = "ClusterIP"
        ports = {
          amqp = 5672
        }
      }
      metrics = {
        enabled = true # CloudWatch/Prometheus 연동 시 활용
      }
    })
  ]

  depends_on = [
    aws_eks_node_group.backend,
    kubernetes_storage_class_v1.gp3,
  ]
}

# Backend App Service(Spring Boot)에서 참조할 Secret
resource "kubernetes_secret_v1" "rabbitmq_connection" {
  metadata {
    name      = "rabbitmq-connection"
    namespace = kubernetes_namespace.forenshield.metadata[0].name
  }

  data = {
    RABBITMQ_HOST     = "rabbitmq.${kubernetes_namespace.forenshield.metadata[0].name}.svc.cluster.local"
    RABBITMQ_PORT     = "5672"
    RABBITMQ_USERNAME = var.rabbitmq_username
    RABBITMQ_PASSWORD = random_password.rabbitmq.result
  }

  type = "Opaque"

  depends_on = [helm_release.rabbitmq]
}
```

### Terraform으로 하지 않는 것

| 항목 | 이유 |
|------|------|
| Amazon MQ | 본 프로젝트는 EKS in-cluster 방식 |
| `sg-rabbitmq` | EKS Node SG 내부 통신, 별도 AWS SG 불필요 |
| Exchange/Queue 선언 | 앱(Spring Boot) 또는 별도 Job에서 처리 — 인프라 Terraform 범위 밖 |

## Step 12 — Application Load Balancer

| 항목 | 설정 |
|------|------|
| ALB | Public Subnet, HTTPS 443 |
| Target Group | Frontend Node → port 80 |
| Listener | 443 → Target Group (ACM 인증서) |
| Health Check | `/health` |

> EKS 환경에서는 **AWS Load Balancer Controller + Ingress**로 ALB를 생성하는 것이 일반적입니다 (`md/5.frontend-deploy.md`).  
> 아래는 **수동 ALB** 생성 CLI 예시입니다.

### AWS CLI (수동 ALB 예시)

```bash
export ALB_ARN=$(aws elbv2 create-load-balancer \
  --name forenshield-alb \
  --subnets $PUB_SUBNET_A $PUB_SUBNET_B \
  --security-groups $SG_ALB \
  --scheme internet-facing \
  --type application \
  --query 'LoadBalancers[0].LoadBalancerArn' --output text)

export TG_ARN=$(aws elbv2 create-target-group \
  --name forenshield-frontend-tg \
  --protocol HTTP --port 80 \
  --vpc-id $VPC_ID \
  --health-check-path /health \
  --health-check-interval-seconds 15 \
  --query 'TargetGroups[0].TargetGroupArn' --output text)

# ACM 인증서 ARN (사전 발급 필요)
export ACM_ARN=arn:aws:acm:ap-northeast-2:${ACCOUNT_ID}:certificate/xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx

aws elbv2 create-listener \
  --load-balancer-arn $ALB_ARN \
  --protocol HTTPS --port 443 \
  --certificates CertificateArn=$ACM_ARN \
  --default-actions Type=forward,TargetGroupArn=$TG_ARN

aws elbv2 describe-load-balancers --load-balancer-arns $ALB_ARN \
  --query 'LoadBalancers[0].DNSName' --output text
```

## Step 13 — Route 53

| 항목 | 설정 |
|------|------|
| Hosted Zone | 프로젝트 도메인 |
| A Record (Alias) | 도메인 → ALB |

### AWS CLI

```bash
export DOMAIN=forenshield.example.com

# Hosted Zone 생성 (또는 기존 zone-id 사용)
export ZONE_ID=$(aws route53 create-hosted-zone \
  --name $DOMAIN \
  --caller-reference forenshield-$(date +%s) \
  --query 'HostedZone.Id' --output text)

# NS 레코드를 도메인 등록기관에 설정
aws route53 get-hosted-zone --id $ZONE_ID \
  --query 'DelegationSet.NameServers' --output table

# ALB Alias A 레코드 (ALB_ARN 생성 후)
export ALB_DNS=$(aws elbv2 describe-load-balancers --load-balancer-arns $ALB_ARN \
  --query 'LoadBalancers[0].DNSName' --output text)
export ALB_ZONE=$(aws elbv2 describe-load-balancers --load-balancer-arns $ALB_ARN \
  --query 'LoadBalancers[0].CanonicalHostedZoneId' --output text)

aws route53 change-resource-record-sets --hosted-zone-id $ZONE_ID --change-batch "{
  \"Changes\": [{
    \"Action\": \"CREATE\",
    \"ResourceRecordSet\": {
      \"Name\": \"$DOMAIN\",
      \"Type\": \"A\",
      \"AliasTarget\": {
        \"HostedZoneId\": \"$ALB_ZONE\",
        \"DNSName\": \"$ALB_DNS\",
        \"EvaluateTargetHealth\": true
      }
    }
  }]
}"
```

### 검증

```bash
dig $DOMAIN
curl -I https://$DOMAIN/health
```

## Step 14 — CloudWatch · SSM

| 항목 | 설정 |
|------|------|
| CloudWatch Log Groups | EKS, ALB, RDS |
| CloudWatch Alarms | CPU, 5xx, RDS 연결 수 |
| SSM Session Manager | Private EC2/EKS Node 접속 |

### AWS CLI — Log Groups · Alarms

```bash
aws logs create-log-group --log-group-name /aws/eks/${CLUSTER_NAME}/cluster
aws logs put-retention-policy --log-group-name /aws/eks/${CLUSTER_NAME}/cluster --retention-in-days 30

aws logs create-log-group --log-group-name /aws/rds/instance/forenshield-db/postgresql
aws logs put-retention-policy --log-group-name /aws/rds/instance/forenshield-db/postgresql --retention-in-days 30

# RDS CPU 알람
aws cloudwatch put-metric-alarm \
  --alarm-name forenshield-rds-cpu-high \
  --metric-name CPUUtilization \
  --namespace AWS/RDS \
  --statistic Average \
  --period 300 \
  --threshold 80 \
  --comparison-operator GreaterThanThreshold \
  --evaluation-periods 2 \
  --dimensions Name=DBInstanceIdentifier,Value=forenshield-db
```

### AWS CLI — SSM (EKS Node 접속)

```bash
# EKS Node EC2 Instance ID 조회
aws ec2 describe-instances \
  --filters "Name=tag:eks:cluster-name,Values=$CLUSTER_NAME" "Name=instance-state-name,Values=running" \
  --query 'Reservations[*].Instances[*].[InstanceId,Tags[?Key==`eks:nodegroup-name`].Value|[0]]' \
  --output table

export INSTANCE_ID=i-xxxxxxxxxxxxxxxxx

aws ssm start-session --target $INSTANCE_ID
```

### 검증

```bash
aws logs describe-log-groups --log-group-name-prefix /aws/eks/forenshield --output table
aws cloudwatch describe-alarms --alarm-names forenshield-rds-cpu-high --output table
```
